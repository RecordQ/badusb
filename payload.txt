REM Advanced Features Test Payload

REM Define a function to wiggle the mouse
FUNCTION wiggle
  MOUSE MOVE 10 10
  DELAY 100
  MOUSE MOVE -10 -10
  DELAY 100
  STRING HI 
END_FUNCTION

REM Loop 5 times
REPEAT 5
  LED ON
  CALL wiggle
  LED OFF
  RANDOM_DELAY 200 800
END_REPEAT
